public sealed class Distribution
{
    public List<Format> Formats { get; set; } = new();          // gmd:distributionFormat
    public List<OnlineResource> OnlineResources { get; set; } = new(); // gmd:onLine
    public string? UseLimitation { get; set; }                  // gmd:useLimitation
    public string? AccessConstraints { get; set; }              // gmd:accessConstraints
}

public sealed class Format
{
    public string Name { get; set; } = string.Empty;            // gmd:MD_Format/gmd:name
    public string? Version { get; set; }                        // gmd:MD_Format/gmd:version
}

public sealed class OnlineResource
{
    public string Url { get; set; } = string.Empty;             // gmd:CI_OnlineResource/gmd:linkage/gmd:URL
    public string? Description { get; set; }                    // gmd:CI_OnlineResource/gmd:description
    public string? Protocol { get; set; }                       // gmd:CI_OnlineResource/gmd:protocol
}

public sealed class DataQuality
{
    public string? Lineage { get; set; }                        // gmd:LI_Lineage/gmd:statement
    public List<ProcessStep> ProcessSteps { get; set; } = new();// gmd:LI_Lineage/gmd:processStep
}

public sealed class ProcessStep
{
    public string? Description { get; set; }                    // gmd:LI_ProcessStep/gmd:description
    public DateTime? Date { get; set; }                         // gmd:LI_ProcessStep/gmd:dateTime
    public Organization? Processor { get; set; }                // gmd:LI_ProcessStep/gmd:processor
    // Organization class already exists in the bundle
}
